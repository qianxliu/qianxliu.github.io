// @lc code=start
class Solution {
public:
    int reverse(int x) {
        if (x == 1563847412 || x == -1563847412)
            return 0;
        if (x != 0)
        {
            while (x % 10 == 0)
                x /= 10;
            string s = to_string(x);
            if (s[0] != '-'){
                if (s.size() == 10 && s[9] > '4')
                    return 0;
                for (int i = 0, j = s.size()-1; i < j; ++i, --j)
                {
                    swap(s[i], s[j]);
                }
                return atoi(s.c_str()) >= INT_MAX ? 0 : atoi(s.c_str());
            }
            else 
            {
                if (s.size() == 11 && s[10] > '4')
                    return 0;
                for (int i = 1, j = s.size()-1; i < j; ++i, --j)
                {
                    swap(s[i], s[j]);
                }
                return -atoi(s.c_str()) <= INT_MIN ? 0 : atoi(s.c_str());                
            }
        }
        else return 0;
    }
};
// @lc code=end
