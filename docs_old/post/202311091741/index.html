class Solution {
public:
    //You need to memo the count of characters
    int lengthOfLongestSubstring(string s) {
        unordered_map<char,int> window;
        int left = 0, right = 0;
        int maxLen = 0;
        while(right < s.size())
        {
            char c = s[right];
            ++right;
            ++window[c];
            while(window[c] > 1)
            {
                ++left;
                --window[c];
            }
            maxLen = max(maxLen, right - left);
        }
        return maxLen;
    }
};
